function countUp(elId, endVal, isPct, prefix="") {

const el = document.getElementById(elId); if(!el) return;

let duration = 3500, start = 0, startTime = null, end = parseFloat(endVal) || 0;

const step = (t) => {

if(!startTime) startTime = t; let p = Math.min((t - startTime)/duration, 1), ease = 1 - Math.pow(1 - p, 3);

let curr = start + ease * (end - start);

el.innerText = prefix + (isPct ? curr.toFixed(1) + '%' : Math.floor(curr));

if(p < 1) requestAnimationFrame(step); else el.innerText = prefix + (isPct ? end.toFixed(1) + '%' : end);

}; requestAnimationFrame(step);

}

function getActivityStatus(ts) {

if (!ts || ts === 0) return { color: "var(--dot-red)", text: "Inactive 24h+" };

const diffHours = (new Date().getTime() - ts) / (1000 * 60 * 60);

if (diffHours < 2) return { color: "var(--dot-green)", text: "Active recently" };

if (diffHours < 24) return { color: "var(--dot-yellow)", text: "Active today" };

return { color: "var(--dot-red)", text: `Active ${Math.floor(diffHours/24)}d ago` };

}

const getRadarGradient = (context, colorStart, colorEnd) => {

const chartArea = context.chart.chartArea; if (!chartArea) return colorStart; const ctx = context.chart.ctx; const centerX = (chartArea.left + chartArea.right) / 2;

const centerY = (chartArea.top + chartArea.bottom) / 2; const r = Math.min(chartArea.right - chartArea.left, chartArea.bottom - chartArea.top) / 2;

const gradient = ctx.createRadialGradient(centerX, centerY, 0, centerX, centerY, r); gradient.addColorStop(0, colorStart); gradient.addColorStop(1, colorEnd); return gradient;

};
